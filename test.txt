import pandas as pd
import numpy as np

array1 = np.array([1, 2, 3])
array2 = np.array([4, 5])
array3 = np.array([6, 7, 8, 9])

# Find the longest array
max_length = max(len(array1), len(array2), len(array3))

# Pad shorter arrays with NaN
array1 = np.pad(array1, (0, max_length - len(array1)), 'constant', constant_values=np.nan)
array2 = np.pad(array2, (0, max_length - len(array2)), 'constant', constant_values=np.nan)
array3 = np.pad(array3, (0, max_length - len(array3)), 'constant', constant_values=np.nan)

# Create DataFrame
df = pd.DataFrame({
    'Column1': array1,
    'Column2': array2,
    'Column3': array3
})

print(df)
